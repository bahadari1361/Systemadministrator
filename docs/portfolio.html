DNS Automation Scripts

# PowerShell: Automated DNS Zone Backup
$Zones = Get-DnsServerZone | Where-Object {$_.ZoneType -eq "Primary"}
foreach ($Zone in $Zones) {
  $FileName = "C:\DNS_Backups\" + $Zone.ZoneName + "_" + (Get-Date -Format "yyyyMMdd") + ".bak"
  Export-DnsServerZone -Name $Zone.ZoneName -FileName $FileName
  Write-EventLog -LogName "DNS" -Source "Automation" -EventId 100 -Message "Backed up zone $($Zone.ZoneName)"
}
