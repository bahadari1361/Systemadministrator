# Automated zone transfer script
$Zones = Get-DnsServerZone | Where-Object {$_.IsAutoCreated -eq $false}
foreach ($Zone in $Zones) {
  Export-DnsServerZone -Name $Zone.ZoneName -FileName "C:\Backups\$($Zone.ZoneName).bak"
  Write-EventLog -LogName 'DNS' -Source 'ZoneTransfer' -EntryType Information -EventId 100 -Message "Backup created for $($Zone.ZoneName)"
}
